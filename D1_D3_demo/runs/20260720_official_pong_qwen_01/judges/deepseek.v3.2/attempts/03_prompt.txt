import sys
import time

# ==================== 常量定义 ====================
SCREEN_WIDTH = 800
SCREEN_HEIGHT = 600
BLACK = (0, 0, 0)
WHITE = (255, 255, 255)
PADDLE_WIDTH = 15
PADDLE_HEIGHT = 100
BALL_SIZE = 20
PADDLE_SPEED = 7
AI_SPEED = 5
BALL_SPEED_INITIAL = 5
BALL_SPEED_INCREMENT = 0.5
MAX_BALL_SPEED = 12
TARGET_SCORE = 5
GAME_DURATION = 120

# 游戏方向常量
UP = -1
DOWN = 1

# ==================== 类定义 ====================

class Paddle:
    def __init__(self, x, y):
        self.rect = pygame.Rect(x, y, PADDLE_WIDTH, PADDLE_HEIGHT)
        self.speed = PADDLE_SPEED
        self.direction = 0  # -1: up, 0: stopped, 1: down

    def move_up(self):
        self.direction = UP

    def move_down(self):
        self.direction = DOWN

    def stop(self):
        self.direction = 0

    def update(self):
        if self.direction == UP:
            self.rect.y -= self.speed
        elif self.direction == DOWN:
            self.rect.y += self.speed

        # 边界检查
        if self.rect.top < 0:
            self.rect.top = 0
        if self.rect.bottom > SCREEN_HEIGHT:
            self.rect.bottom = SCREEN_HEIGHT

    def draw(self, screen):
        pygame.draw.rect(screen, WHITE, self.rect)

class Ball:
    def __init__(self):
        self.reset()

    def reset(self):
        self.rect = pygame.Rect(SCREEN_WIDTH // 2 - BALL_SIZE // 2,
                               SCREEN_HEIGHT // 2 - BALL_SIZE // 2,
                               BALL_SIZE, BALL_SIZE)
        # 随机选择发球方向
        direction_x = 1 if time.time() % 2 == 0 else -1
        self.speed_x = BALL_SPEED_INITIAL * direction_x
        self.speed_y = BALL_SPEED_INITIAL * (2 * (time.time() % 1) - 1)  # 约0左右的y方向
        # 确保y方向速度不为0
        if abs(self.speed_y) < 1:
            self.speed_y = BALL_SPEED_INITIAL * 0.8
        self.speed_multiplier = 1

    def update(self):
        # 更新位置
        self.rect.x += self.speed_x
        self.rect.y += self.speed_y

        # 上下边界反弹
        if self.rect.top <= 0 or self.rect.bottom >= SCREEN_HEIGHT:
            self.speed_y = -self.speed_y

    def increase_speed(self):
        self.speed_multiplier = min(self.speed_multiplier + BALL_SPEED_INCREMENT, MAX_BALL_SPEED)
        # 同时增加x和y方向的速度分量
        self.speed_x = self.speed_x / abs(self.speed_x) * self.speed_multiplier
        self.speed_y = self.speed_y / abs(self.speed_y) * self.speed_multiplier

    def draw(self, screen):
        pygame.draw.ellipse(screen, WHITE, self.rect)
